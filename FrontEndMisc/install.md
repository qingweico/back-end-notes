```shell
#  安装 windows C++ 构建工具
npm install --global windows-build-tools
# 安装脚手架 
npm install -g @vue/cli
# npm 淘宝镜像
npm install --registry=https://registry.npm.taobao.org
# NVM for Windows
https://github.com/coreybutler/nvm-windows
# 查看可用的 Node.js 版本
nvm list available
# 安装特定的 Node.js 版本
nvm install ${version}
# 切换到已安装的 Node.js 版本
nvm use ${version}
```
